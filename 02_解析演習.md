### 2.　BWAでマッピング & samtoolsでソート

02_bwa.sh
```
#!/bin/bash
set -euo pipefail


# ===== 引数のチェック =====
if [ "$#" -ne 5 ]; then
    echo "Usage: $0 <SAMPLE> <RUNID> <R1.fastq.gz> <R2.fastq.gz> <REFERENCE.fasta>"
    exit 1
fi

# ===== 引数の取得 =====
SAMPLE="$1"
RUNID="$2"
R1="$3"
R2="$4"
REF="$5"

# ===== 設定 =====
THREADS=8
OUT="${SAMPLE}.sorted.bam"
LOG="align_${SAMPLE}.log"

# ===== 実行=====
echo "Start alignment: $(date)" | tee "${LOG}"

bwa mem -t "${THREADS}" \
    -R "@RG\tID:${RUNID}\tSM:${SAMPLE}\tPL:ILLUMINA" \
    "${REF}" \
    "${R1}" "${R2}" 2>> "${LOG}" | \
    samtools sort -@ "${THREADS}" -o "${OUT}" 2>> "${LOG}"

samtools index "${OUT}" 2>> "${LOG}"

echo "Finished: $(date)" | tee -a "${LOG}" 
```

### 3.　MarkDuplicates で 重複リードを検出

03_markduplicates.sh
```
#!/bin/bash
set -euo pipefail

# ===== 引数チェック =====
if [ "$#" -ne 1 ]; then
echo "Usage: $0 SAMPLE_NAME"
echo "Example: $0 NA18939"
exit 1
fi

# ===== 設定 =====
THREADS=8
SAMPLE=$1

IN_BAM="${SAMPLE}.sorted.bam"
OUT_BAM="${SAMPLE}.dedup.bam"
METRICS="${SAMPLE}.dup_metrics.txt"
LOG="markdup_${SAMPLE}.log"

# ===== 開始 =====
echo "Start MarkDuplicates: $(date)" | tee "$LOG"

# ===== 実行 =====
gatk MarkDuplicates \
-I "$IN_BAM" \
-O "$OUT_BAM" \
-M "$METRICS" \
2>> "$LOG"

# ===== インデックス作成 =====
samtools index "$OUT_BAM" 2>> "$LOG"

# ===== 終了 =====
echo "Finished MarkDuplicates: $(date)" | tee -a "$LOG"
```

```
./03_markduplicates.sh NA18939
```

04_BR.sh
```
#!/bin/bash
set -euo pipefail


# ===== 引数チェック =====
if [ $# -ne 1 ]; then
    echo "Usage: $0 SAMPLE_NAME"
    exit 1
fi

# ===== 設定 =====
SAMPLE=$1

REF=~/Desktop/DICP/20260403_GATK_brew/reference/hg38/Homo_sapiens_assembly38.fasta
DBSNP=~/Desktop/DICP/20260403_GATK_brew/reference/hg38/dbsnp_146.hg38.vcf.gz
MILLS=~/Desktop/DICP/20260403_GATK_brew/reference/hg38/Mills_and_1000G_gold_standard.indels.hg38.vcf.gz

IN="${SAMPLE}.dedup.bam"
OUT="${SAMPLE}.recal.table"
LOG="baserecal_${SAMPLE}.log"

# ===== 入力ファイル確認 =====
for file in "$REF" "$DBSNP" "$MILLS" "$IN"; do
    if [ ! -f "$file" ]; then
        echo "Error: File not found -> $file"
        exit 1
    fi
done

# ===== 開始 =====
echo "Start BaseRecalibrator: $(date)" | tee "$LOG"

# ===== 実行 =====
gatk BaseRecalibrator \
    -R "$REF" \
    -I "$IN" \
    --known-sites "$DBSNP" \
    --known-sites "$MILLS" \
    -O "$OUT" \
    2>> "$LOG"

# ===== 終了 =====
echo "Finished BaseRecalibrator: $(date)" | tee -a "$LOG"
```
実行
```
./04_BR.sh NA18939
```

### 5.　ApplyBQSR で BaseRecalibratorで算出した補正テーブルを用いて塩基品質スコアを再調整

```
#!/bin/bash
set -euo pipefail

# ===== 引数チェック =====
if [ $# -ne 1 ]; then
    echo "Usage: $0 SAMPLE_NAME"
    exit 1
```

### 6.　HaplotypeCaller で 変異　(variant)検出
VCF（変異リスト）出す

06_haplotypecaller.sh
```
#!/bin/bash
set -euo pipefail

# ===== 引数チェック =====
if [ $# -ne 1 ]; then
    echo "Usage: $0 SAMPLE_NAME"
    exit 1
fi

# ===== 設定 =====
SAMPLE=$1 
THREADS=8

REF=~/Desktop/DICP/20260403_GATK_brew/reference/hg38/Homo_sapiens_assembly38.fasta
IN_BAM="${SAMPLE}.recal.bam"
OUT="${SAMPLE}.g.vcf.gz"
LOG="haplotypecaller_${SAMPLE}.log"

# ===== 入力ファイル確認 =====
for file in "$REF" "$IN_BAM"; do
    if [ ! -f "$file" ]; then
        echo "Error: File not found -> $file"
        exit 1
    fi
done

# ===== 開始 =====
echo "HaplotypeCaller: $(date)" | tee "$LOG"

# ===== 実行 ===== 
gatk --java-options "-Xmx16G" HaplotypeCaller \
-R "$REF" \
-I "$IN_BAM" \
-O "$OUT_GVCF" \
-ERC GVCF \
--native-pair-hmm-threads "$THREADS" \
2>> "$LOG"

# ===== インデックス作成 =====
gatk IndexFeatureFile -I "$OUT_GVCF" 2>> "$LOG"

# ===== 終了 =====
echo "Finished HaplotypeCaller: $(date)" | tee -a "$LOG"

```
実行
